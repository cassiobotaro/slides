           Field('slug'),
           Field('conteudo', 'text'),
           Field('publicado', 'boolean'),
           Field('tags', 'list:string'),
           Field('categoria', 'list:reference categoria'),
           auth.signature
        )
        db.post.slug.compute = lambda registro:IS_SLUG()(registro.titulo)[0]
        db.post.categoria.requires = IS_IN_DB(db, 'categoria.id', '%(nome)s', multiple=True)

        def show():
            post_slug = request.args(0, otherwise=URL('default', 'index'))
            # atalho para db(db.post.slug == post_slug).select().first()
            post = db.post(slug=post_slug, publicado=True)
            if not post:
                raise HTTP(404)
            return {'post':post}
